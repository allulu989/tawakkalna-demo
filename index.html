// ============ SCREEN NAVIGATION ============

function goToScreen(screen) {

    if (animInterval) { clearInterval(animInterval); animInterval = null; }

    currentScreen = screen;

    updateStepDots();

    updateInstruction();

 

    switch(screen) {

        case 0: showOldScreen('old-s1'); hideNewPhone(); break;

        case 1: showOldScreen('old-s2'); hideNewPhone(); break;

        case 2: showOldScreen('old-s3'); hideNewPhone(); startPreparing(); break;

        case 3: showOldScreen('old-s4'); showNewPhone(); showNewScreen('new-s4'); resetScan(); break;

        case 4: showOldScreen('old-s5'); showNewScreen('new-s5'); startTransferring(); break;

        case 5: showOldScreen('old-s6'); showNewScreen('new-s6'); launchConfetti(); break;

    }

}

 

function showOldScreen(id) {

    document.querySelectorAll('#oldPhoneWrap .screen').forEach(s => s.classList.remove('active'));

    document.getElementById(id).classList.add('active');

}

 

function showNewScreen(id) {

    document.querySelectorAll('#newPhoneWrap .screen').forEach(s => s.classList.remove('active'));

    document.getElementById(id).classList.add('active');

}

 

function showNewPhone() {

    const el = document.getElementById('newPhoneWrap');

    el.classList.remove('hidden-phone');

    el.classList.add('visible-phone');

}

 

function hideNewPhone() {

    const el = document.getElementById('newPhoneWrap');

    el.classList.remove('visible-phone');

    el.classList.add('hidden-phone');

}

 

// ============ STEP DOTS ============

function updateStepDots() {

    document.querySelectorAll('.step-dot').forEach((dot, i) => {

        dot.classList.remove('active', 'done');

        if (i < currentScreen) dot.classList.add('done');

        else if (i === currentScreen) dot.classList.add('active');

    });

}

 

function updateInstruction() {

    const el = document.getElementById('instrText');

    const hint = currentScreen === 2 || currentScreen === 4 ? '' : '<span class="click-hint">👆 اضغط</span>';

    el.innerHTML = instructions[currentScreen] + ' ' + hint;

}

function startPreparing() {

    const fill = document.getElementById('old-s3-fill');

    const icon1 = document.getElementById('old-s3-i1');

    const text1 = document.getElementById('old-s3-t1');

   

    // Reset

    fill.style.width = '0%';

    icon1.className = 'stp-icon active';

    icon1.innerHTML = '<span class="spin">⟳</span>';

    text1.className = '';

   

    let pct = 0;

    animInterval = setInterval(() => {

        pct += 2;

        if (pct > 100) pct = 100;

        fill.style.width = pct + '%';

        // Arabic numeral conversion

        lbl.textContent = 'جاري التحضير: ' + toAr(pct) + '٪';

        if (pct >= 100) {

            clearInterval(animInterval);

            animInterval = null;

            icon1.className = 'stp-icon done';

            icon1.innerHTML = '<span>✓</span>';

            text1.className = 'done-t';

            setTimeout(() => goToScreen(3), 600);

        }

    }, 60);

}

    const colors = ['#1A6B54','#2EC4A0','#F59E0B','#3B7DDD','#7C3AED','#EC4899','#fff'];

    for (let i = 0; i < 80; i++) {

        const piece = document.createElement('div');

        piece.className = 'confetti-piece';

        piece.style.left = Math.random() * 100 + '%';

        piece.style.background = colors[Math.floor(Math.random() * colors.length)];

        piece.style.width = (Math.random() * 8 + 4) + 'px';

        piece.style.height = (Math.random() * 8 + 4) + 'px';

        piece.style.borderRadius = Math.random() > 0.5 ? '50%' : '2px';

        piece.style.animationDuration = (Math.random() * 2 + 2) + 's';

        piece.style.animationDelay = (Math.random() * 1) + 's';

        container.appendChild(piece);

    }

    setTimeout(() => { container.innerHTML = ''; }, 5000);

}

    document.getElementById('confetti').innerHTML = '';

    hideNewPhone();

    document.querySelectorAll('#newPhoneWrap .screen').forEach(s => s.classList.remove('active'));

    document.getElementById('new-s4').classList.add('active');

    // Reset preparing screen

    document.getElementById('old-s3-fill').style.width = '0%';

    document.getElementById('old-s3-lbl').textContent = 'جاري التحضير: ٠٪';

document.addEventListener('keydown', (e) => {

    if (e.key === 'ArrowLeft' || e.key === ' ') {

        e.preventDefault();

        // Auto advance for non-auto screens

        if (currentScreen === 0) goToScreen(1);

        else if (currentScreen === 1) goToScreen(2);

        else if (currentScreen === 3) startScan();

    }

    if (e.key === 'r' || e.key === 'R') restart();

});
